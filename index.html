from flask import Flask, request, jsonify, render_template_string import re import math app = Flask(__name__) # --- ЛОГИКА АНАЛИЗА (МАКСИМАЛЬНАЯ СКОРОСТЬ) --- def analyze_text_logic(text): sentences = re.split(r'[.!?]+', text) sentences = [s.strip() for s in sentences if len(s.strip()) > 5] words = re.findall(r'\w+', text.lower()) if len(words) < 10: return 0, 100 # Слишком мало текста для вывода # 1. Анализ вариативности предложений (Люди пишут контрастно) sentence_lengths = [len(s.split()) for s in sentences] if len(sentence_lengths) > 1: avg_len = sum(sentence_lengths) / len(sentence_lengths) variance = sum((x - avg_len) ** 2 for x in sentence_lengths) / len(sentence_lengths) else: variance = 0 # 2. Анализ уникальности слов unique_ratio = len(set(words)) / len(words) # Веса для определения ИИ # ИИ: низкая вариативность (ритм) и средняя уникальность слов ai_score = 40 if variance < 12: ai_score += 30 # Монотонный ритм характерный для ИИ if unique_ratio < 0.45: ai_score += 20 # Слишком простой/повторяющийся лексикон # Корректировка границ ai_final = min(max(int(ai_score), 5), 96) return ai_final, 100 - ai_final # --- ДИЗАЙН (HIGH-END MINIMALISM) --- HTML_TEMPLATE = ''' AI Detector Elite

AI DETECTOR

0% AI Content
0% Human Text
''' @app.route('/') def index(): return render_template_string(HTML_TEMPLATE) @app.route('/analyze', methods=['POST']) def analyze(): text = request.json.get('text', '') ai, human = analyze_text_logic(text) return jsonify({'ai': ai, 'human': human}) if __name__ == '__main__': app.run(debug=True)
