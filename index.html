from flask import Flask, request, jsonify, render_template_string import re app = Flask(__name__) # --- HTML И ДИЗАЙН (Все в одном) --- HTML_TEMPLATE = ''' AI vs Human Detector

AI Detector

Написано ИИ:0%
Человек:0%
''' # --- ЛОГИКА АНАЛИЗА --- def get_score(text): words = text.split() # Имитация работы ИИ детектора (базируется на длине слов и сложности) # В реальности тут подключается модель Hugging Face avg_len = sum(len(w) for w in words) / len(words) if words else 0 # Нейросети часто пишут очень "средними" по длине словами ai_prob = 50 + (avg_len * 2) if avg_len < 7 else 30 ai_prob = min(max(int(ai_prob), 5), 95) # Ограничиваем от 5 до 95% return ai_prob, 100 - ai_prob @app.route('/') def home(): return render_template_string(HTML_TEMPLATE) @app.route('/analyze', methods=['POST']) def analyze(): content = request.json.get('text', '') ai, human = get_score(content) return jsonify({'ai': ai, 'human': human}) if __name__ == '__main__': app.run(debug=True)
